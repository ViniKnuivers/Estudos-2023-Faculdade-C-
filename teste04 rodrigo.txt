#include <iostream>
#include <string.h>

using namespace std;

void inverte(char nome[])
{
    int tam = strlen(nome);
    for (int i = tam - 1; i >= 0; i--)
    {
      cout << nome[i];
    }
}

{
    char nome[] = "Vinicius Knuivers";
    inverte(nome);
